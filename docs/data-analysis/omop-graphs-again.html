with pg.connect(db_uri) as conn:
  for relationship in conn.execute("""
  SELECT relationship_id, COUNT(concept_id_1) as count
  FROM cdm.concept_relationship
  GROUP BY relationship_id
  ORDER BY count DESC;
  """).fetchall():
    print(relationship)
